#Instance Template
gcloud beta compute --project=pe-training instance-templates create varun-pe-it --machine-type=n1-standard-1 --network=projects/pe-training/global/networks/default --network-tier=PREMIUM --service-account=912623308461-compute@developer.gserviceaccount.com --image=debian-9-stretch-v20180716 --image-project=debian-cloud --boot-disk-size=10GB --boot-disk-type=pd-standard
#Instance Group
gcloud compute --project "pe-training" instance-groups managed create "varun-pe-ig" --base-instance-name "varun-pe-ig" --template "varun-pe-it" --size "1" --zone "us-east1-b"

gcloud compute --project "pe-training" instance-groups managed set-autoscaling "varun-pe-ig" --zone "us-east1-b" --cool-down-period "60" --max-num-replicas "10" --min-num-replicas "1" --target-cpu-utilization "0.6"

#Health Check
gcloud compute --project "pe-training" http-health-checks create "varun-pe-hc" --port "80" --request-path "/" --check-interval "5" --timeout "5" --unhealthy-threshold "2" --healthy-threshold "2"

#named port
gcloud compute --project "pe-training" instance-groups set-named-ports my-instance-group --named-ports http-port:80

gcloud compute --project "pe-training" backend-services create my-http-backend-service
--http-health-checks my-healthcheck --port-name http-port --protocol HTTP

gcloud compute --project "pe-training" backend-services add-backend my-http-backend-service
--instance-group my-instance-group --balancing-mode RATE
--max-rate-per-instance 10
